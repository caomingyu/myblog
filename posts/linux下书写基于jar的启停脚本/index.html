在Linux系统上，怎么书写jar的启停脚本

在Linux系统上，怎么书写jar的启停脚本

  • 启动脚本参考
  #!/bin/sh
  pidfile=tpid
  if [ -f "$pidfile" ] && kill -0 $( cat "$pidfile" ); then
  echo "jar 已经在运行中..."
  exit 1
  fi
  nohup java -jar <jar包名> --spring.config.location=/home/cmy/test/ --logging.config=logback-spring.xml >/dev/null 2>&1&
  echo $! > tpid
  echo "jar启动成功"
  • 停止脚本参考
  #!/bin/sh
  pidfile=tpid
  if [ ! -f "$pidfile" ] || ! kill -0 $( cat "$pidfile" ); then
  echo "jar没有在运行"
  else
  echo "正在停止jar..."
  PID=$(cat "$pidfile")
  kill -9 $PID
  rm "$pidfile" && echo "jar已停止"
  fi
